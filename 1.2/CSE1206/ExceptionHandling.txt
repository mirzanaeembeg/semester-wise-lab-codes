package Question06;

public class Registration {

    String name;

    String password;

    Registration() {
    }

    Registration(String userName, String password) {

        this.name = name;

        int i = 0;
        try {
            if (password.length() > 8) {
                i++;
            } else {

                throw new LengthException("Password must be 8 character long");
            }
        } catch (LengthException e) {
            System.out.println(e);
        }

        try {
            if (password.matches(".*[A-Z]*.")) {
                i++;
            } else {

                throw new UpperCaseException("Password Must be one UpperCase Letter");

            }
        } catch (UpperCaseException e) {
            System.out.println(e);
        }

        try {
            if (password.matches(".*[a-z]*.")) {
                i++;
            } else {

                throw new LowerCaseException("Password must be one LowerCase letter");
            }
        } catch (LowerCaseException e) {
            System.out.println(e);
        }

        if (i == 3) {
            this.password = password;
        } else {
            this.password = null;

        }

    }

}

package Question06;

public class UpperCaseException extends Exception {

    UpperCaseException() {
    }

    UpperCaseException(String msg) {
        super(msg);

    }

}

public class LengthException extends Exception {

    LengthException() {
    }

    LengthException(String msg) {
        super(msg);

    }

}

    LowerCaseException() {
    }

        super(msg);

    }

}
